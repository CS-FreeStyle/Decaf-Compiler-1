19 Function *methodDeclaration::generateCode(globals *currentGlobals) {
20  std::vector<std::string> argNames;
21  std::vector<std::string> argTypes;
22  std::vector<class methodArgument *> args = arg_list->getArgList();
23  std::vector<Type *> arguments;
24  int arg_size = args.size();
25  for (int i = 0; i < args.size(); i++) {
26  /* Iterate over the arguments and get the types of them in llvm */
27  std::string arg_type = args[i]->getType();
28  std::string arg_name = args[i]->getName();
29  if (arg_type == "int") {
30  arguments.push_back(Type::getInt32Ty(currentGlobals->Context));
31  } else if (arg_type == "boolean") {
32  arguments.push_back(Type::getInt1Ty(currentGlobals->Context));
33  } else {
34  currentGlobals->errors++;
35  reportError("Arguments can only be int or boolean");
36  return 0;
37  }
38  argTypes.emplace_back(arg_type);
39  argNames.emplace_back(arg_name);
40  }
41 
42  Type *returnType;
43  /* Get the return Type */
44  if (type == "int") {
45  returnType = Type::getInt32Ty(currentGlobals->Context);
46  } else if (type == "boolean") {
47  returnType = Type::getInt1Ty(currentGlobals->Context);
48  } else if (type == "void") {
49  returnType = Type::getVoidTy(currentGlobals->Context);
50  } else {
51  currentGlobals->errors++;
52  reportError("Invalid Return Type for " + name + ". Return Type can only be int or boolean or bool");
53  return 0;
54  }
55 
56  /* Get the function type and create a Function */
57  FunctionType *FT = llvm::FunctionType::get(returnType, arguments, false);
58  Function *F = llvm::Function::Create(FT, Function::ExternalLinkage, name, currentGlobals->TheModule);
59 
60  /* Iterate through arguments and set the Names for them */
61 
62  unsigned Idx = 0;
63  for (Function::arg_iterator AI = F->arg_begin(); Idx != arg_size; ++AI, ++Idx) {
64  AI->setName(argNames[Idx]);
65  }
66 
67  /* Create a New block for this Function */
68  BasicBlock *BB = BasicBlock::Create(currentGlobals->Context, "entry", F);
69  currentGlobals->Builder->SetInsertPoint(BB);
70  Idx = 0;
71 
72  /* Allocate memory for the arguments passed */
73  for (auto &Arg : F->args()) {
74  if (Idx == arg_size) { break; }
75  AllocaInst *Alloca = currentGlobals->CreateEntryBlockAlloca(F, argNames[Idx], argTypes[Idx]);
76  currentGlobals->Builder->CreateStore(&Arg, Alloca);
77 
78  currentGlobals->NamedValues[argNames[Idx]] = Alloca;
79  Idx++;
80  }
81 
82  Value *RetVal = body->generateCode(currentGlobals);
83  if (RetVal) {
84  /* make this the return value */
85  if (type != "void")
86  currentGlobals->Builder->CreateRet(RetVal);
87  else
88  currentGlobals->Builder->CreateRetVoid();
89  /* Verify the function */
90  verifyFunction(*F);
91  return F;
92  }
93  /* Error Condition */
94  F->eraseFromParent();
95  return 0;
96 }
